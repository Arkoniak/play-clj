(defn add-body!
  [screen body]
  (cond
    (isa? (type (:object body)) btRigidBody)
    (bullet-3d! screen :add-rigid-body (:object body))
    (isa? (type (:object body)) btSoftBody)
    (bullet-3d! screen :add-soft-body (:object body)))
  body)
